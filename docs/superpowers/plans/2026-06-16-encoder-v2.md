import math
import torch
import torch.nn as nn

from loom.training.dataset import (
    CONTINUOUS_KEYS, CATEGORICAL_KEYS, N_CONTINUOUS, N_CATEGORICAL, N_ROUTING, N_PARAMS,
)

try:
    from mamba_ssm import Mamba
    HAS_MAMBA = True
except ImportError:
    HAS_MAMBA = False

# ── Parameter group definitions ─────────────────────────────────────
# Each group: (name, continuous_indices, categorical_keys, n_queries)
# Indices refer to positions within the CONTINUOUS_KEYS list.

OSC_CONT = [i for i, k in enumerate(CONTINUOUS_KEYS)
            if k in ("osc_pitch", "osc_detune", "wt_position",
                      "fm_carrier_ratio", "fm_mod_ratio", "fm_mod_index")]
OSC_CAT = [("osc_waveform", 4), ("osc_type", 3)]

FILTER_CONT = [i for i, k in enumerate(CONTINUOUS_KEYS)
               if k in ("filter_cutoff", "filter_q", "filter_mix",
                         "filt_env_attack", "filt_env_decay", "filt_env_sustain",
                         "filt_env_release", "filt_env_amount")]
FILTER_CAT = [("filter_type", 3)]

ENV_CONT = [i for i, k in enumerate(CONTINUOUS_KEYS)
            if k in ("amp_attack", "amp_decay", "amp_sustain", "amp_release")]

FX_CONT = [i for i, k in enumerate(CONTINUOUS_KEYS)
           if k in ("dist_amount", "dist_mix",
                     "comp_threshold", "comp_ratio", "comp_attack", "comp_release",
                     "comp_makeup", "comp_mix",
                     "chorus_rate", "chorus_depth", "chorus_mix",
                     "delay_time", "delay_feedback", "delay_mix",
                     "reverb_room_size", "reverb_decay", "reverb_damping", "reverb_mix",
                     "eq_low_gain", "eq_mid_gain", "eq_high_gain")]

GLOBAL_CONT = [i for i, k in enumerate(CONTINUOUS_KEYS)
               if k in ("master_gain", "lfo_rate", "lfo_depth", "lfo_phase")]
GLOBAL_CAT = [("lfo_waveform", 4), ("lfo_target", 4)]

PARAM_GROUPS = [
    {"name": "osc",    "cont": OSC_CONT,    "cat": OSC_CAT,    "n_queries": 2, "loss_weight": 1.0},
    {"name": "filter", "cont": FILTER_CONT,  "cat": FILTER_CAT, "n_queries": 2, "loss_weight": 2.0},
    {"name": "env",    "cont": ENV_CONT,     "cat": [],          "n_queries": 2, "loss_weight": 3.0},
    {"name": "fx",     "cont": FX_CONT,      "cat": [],          "n_queries": 2, "loss_weight": 1.0},
    {"name": "global", "cont": GLOBAL_CONT,  "cat": GLOBAL_CAT,  "n_queries": 1, "loss_weight": 1.0},
    {"name": "routing","cont": [],            "cat": [],          "n_queries": 1, "loss_weight": 0.5},
]

N_QUERIES = sum(g["n_queries"] for g in PARAM_GROUPS)  # 10


# ── 2D CNN Stem ─────────────────────────────────────────────────────

class CNN2DStem(nn.Module):
    """4-layer 2D CNN that treats mel spectrogram as a (1, 128, T) image.

    Output: (batch, seq_len, d_model) where seq_len = (128/8) * (T/8).
    """

    def __init__(self, d_model=256):
        super().__init__()
        self.layers = nn.Sequential(
            nn.Conv2d(1, 32, 3, padding=1), nn.BatchNorm2d(32), nn.GELU(),
            nn.Conv2d(32, 64, 3, stride=2, padding=1), nn.BatchNorm2d(64), nn.GELU(),
            nn.Conv2d(64, 128, 3, stride=2, padding=1), nn.BatchNorm2d(128), nn.GELU(),
            nn.Conv2d(128, d_model, 3, stride=2, padding=1), nn.BatchNorm2d(d_model), nn.GELU(),
        )
        self.d_model = d_model

    def forward(self, mel: torch.Tensor) -> torch.Tensor:
        # mel: (batch, n_mels, T)
        x = mel.unsqueeze(1)               # (batch, 1, 128, T)
        x = self.layers(x)                 # (batch, d_model, H, W)
        b, d, h, w = x.shape
        x = x.reshape(b, d, h * w)         # (batch, d_model, seq_len)
        return x.transpose(1, 2)           # (batch, seq_len, d_model)


# ── Learnable 2D Positional Encoding ────────────────────────────────

class LearnedPosEncoding2D(nn.Module):
    def __init__(self, d_model, max_h=16, max_w=32):
        super().__init__()
        self.freq_embed = nn.Embedding(max_h, d_model)
        self.time_embed = nn.Embedding(max_w, d_model)
        self.max_h = max_h
        self.max_w = max_w

    def forward(self, h: int, w: int) -> torch.Tensor:
        freq_pos = self.freq_embed(torch.arange(h, device=self.freq_embed.weight.device))
        time_pos = self.time_embed(torch.arange(w, device=self.time_embed.weight.device))
        # (h, d) + (w, d) → (h*w, d) via broadcasting
        pos = freq_pos.unsqueeze(1) + time_pos.unsqueeze(0)  # (h, w, d)
        return pos.reshape(h * w, -1)  # (seq_len, d)


# ── DETR-style Decoder ──────────────────────────────────────────────

class DETRDecoder(nn.Module):
    """Cross-attention decoder with learnable query tokens."""

    def __init__(self, d_model=256, nhead=8, n_layers=2, n_queries=N_QUERIES):
        super().__init__()
        self.queries = nn.Parameter(torch.randn(n_queries, d_model) * 0.02)
        decoder_layer = nn.TransformerDecoderLayer(
            d_model=d_model, nhead=nhead, dim_feedforward=d_model * 4,
            dropout=0.1, activation="gelu", batch_first=True,
        )
        self.decoder = nn.TransformerDecoder(decoder_layer, num_layers=n_layers)

    def forward(self, memory: torch.Tensor) -> torch.Tensor:
        batch = memory.shape[0]
        queries = self.queries.unsqueeze(0).expand(batch, -1, -1)
        return self.decoder(queries, memory)  # (batch, n_queries, d_model)


# ── Grouped Parameter Heads ─────────────────────────────────────────

class GroupedParamHeads(nn.Module):
    """Per-group heads that read from query tokens and produce the full param vector."""

    def __init__(self, d_model=256):
        super().__init__()
        self.heads = nn.ModuleDict()
        self.group_meta = []
        query_offset = 0

        for g in PARAM_GROUPS:
            name = g["name"]
            n_q = g["n_queries"]
            n_cont = len(g["cont"])
            n_cat = sum(n for _, n in g["cat"])
            n_extra = N_ROUTING if name == "routing" else 0
            n_out = n_cont + n_cat + n_extra

            if n_out > 0:
                self.heads[name] = nn.Sequential(
                    nn.Linear(d_model * n_q, d_model),
                    nn.GELU(),
                    nn.Linear(d_model, n_out),
                )

            self.group_meta.append({
                "name": name,
                "query_slice": (query_offset, query_offset + n_q),
                "cont_indices": g["cont"],
                "cat_groups": g["cat"],
                "n_routing": n_extra,
            })
            query_offset += n_q

    def forward(self, query_out: torch.Tensor) -> torch.Tensor:
        batch = query_out.shape[0]
        continuous = torch.zeros(batch, N_CONTINUOUS, device=query_out.device)
        categoricals = []
        routing = torch.zeros(batch, N_ROUTING, device=query_out.device)

        cat_offset = 0
        cat_list_built = False
        # Pre-build categorical placeholder
        for _, n in CATEGORICAL_KEYS:
            categoricals.append(torch.zeros(batch, n, device=query_out.device))

        for meta in self.group_meta:
            name = meta["name"]
            if name not in self.heads:
                continue
            q_start, q_end = meta["query_slice"]
            q = query_out[:, q_start:q_end, :].reshape(batch, -1)  # concat queries
            raw = self.heads[name](q)

            offset = 0
            # Continuous params → sigmoid
            for idx in meta["cont_indices"]:
                continuous[:, idx] = torch.sigmoid(raw[:, offset])
                offset += 1

            # Categorical params → softmax
            cat_key_idx = 0
            for cat_key, n_cls in meta["cat_groups"]:
                logits = raw[:, offset:offset + n_cls]
                probs = torch.softmax(logits, dim=-1)
                # Find position in CATEGORICAL_KEYS
                for ci, (ck, cn) in enumerate(CATEGORICAL_KEYS):
                    if ck == cat_key:
                        categoricals[ci] = probs
                        break
                offset += n_cls

            # Routing logits (raw, no activation)
            if meta["n_routing"] > 0:
                routing = raw[:, offset:offset + N_ROUTING]

        return torch.cat([continuous] + categoricals + [routing], dim=1)

# ── Architecture A: Transformer Encoder ─────────────────────────────

class TransformerParamEncoder(nn.Module):
    """SynthRL/DETR-style: 2D CNN → Transformer encoder → DETR decoder → grouped heads."""

    def __init__(self, n_mels=128, d_model=256, nhead=8, n_enc_layers=6,
                 n_dec_layers=2, dropout=0.1):
        super().__init__()
        self.stem = CNN2DStem(d_model)
        self.pos_enc = LearnedPosEncoding2D(d_model, max_h=n_mels // 8, max_w=256 // 8)

        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model, nhead=nhead, dim_feedforward=d_model * 4,
            dropout=dropout, activation="gelu", batch_first=True,
        )
        self.encoder = nn.TransformerEncoder(encoder_layer, num_layers=n_enc_layers)
        self.decoder = DETRDecoder(d_model, nhead, n_dec_layers)
        self.heads = GroupedParamHeads(d_model)
        self.d_model = d_model

    def forward(self, mel: torch.Tensor) -> torch.Tensor:
        x = self.stem(mel)  # (batch, seq_len, d_model)

        # Add 2D positional encoding
        _, T_orig = mel.shape[1], mel.shape[2]
        h, w = T_orig // 8 + (1 if T_orig % 8 else 0), 0
        # Infer h, w from stem output
        seq_len = x.shape[1]
        h = 128 // 8  # always 16 after 3 stride-2 layers on 128 mels
        w = seq_len // h
        pos = self.pos_enc(h, w).unsqueeze(0)  # (1, seq_len, d)
        if pos.shape[1] >= seq_len:
            x = x + pos[:, :seq_len]
        else:
            x = x + pos[:, :seq_len]

        memory = self.encoder(x)                # (batch, seq_len, d_model)
        query_out = self.decoder(memory)         # (batch, n_queries, d_model)
        return self.heads(query_out)             # (batch, N_PARAMS)

    def param_groups_for_optimizer(self, lr: float, lr_stem_factor: float = 0.3):
        return [
            {"params": self.stem.parameters(), "lr": lr * lr_stem_factor},
            {"params": list(self.encoder.parameters()) + list(self.pos_enc.parameters()), "lr": lr},
            {"params": list(self.decoder.parameters()) + list(self.heads.parameters()), "lr": lr},
        ]
```

# ── Attention block (for hybrid backbone) ───────────────────────────

class AttentionBlock(nn.Module):
    def __init__(self, d_model, nhead=8, dropout=0.1):
        super().__init__()
        self.norm1 = nn.LayerNorm(d_model)
        self.attn = nn.MultiheadAttention(d_model, nhead, dropout=dropout, batch_first=True)
        self.norm2 = nn.LayerNorm(d_model)
        self.ff = nn.Sequential(
            nn.Linear(d_model, d_model * 4),
            nn.GELU(),
            nn.Dropout(dropout),
            nn.Linear(d_model * 4, d_model),
            nn.Dropout(dropout),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        h = self.norm1(x)
        h, _ = self.attn(h, h, h)
        x = x + h
        x = x + self.ff(self.norm2(x))
        return x


# ── Mamba block (reused from encoder.py, adapted for d_model=256) ──

class MambaBlock(nn.Module):
    def __init__(self, d_model, d_state=64, d_conv=4, expand=2, dropout=0.1):
        super().__init__()
        self.norm1 = nn.LayerNorm(d_model)
        if HAS_MAMBA:
            self.seq = Mamba(d_model=d_model, d_state=d_state, d_conv=d_conv, expand=expand)
        else:
            # S4D fallback for GPUs without mamba-ssm
            from loom.training.encoder import S4DLayer
            self.seq = S4DLayer(d_model, d_state)
        self.use_mamba = HAS_MAMBA
        self.norm2 = nn.LayerNorm(d_model)
        self.ff = nn.Sequential(
            nn.Linear(d_model, d_model * 4),
            nn.GELU(),
            nn.Dropout(dropout),
            nn.Linear(d_model * 4, d_model),
            nn.Dropout(dropout),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        h = self.norm1(x)
        if self.use_mamba:
            h = self.seq(h)
        else:
            h = self.seq(h.transpose(1, 2)).transpose(1, 2)
        x = x + h
        x = x + self.ff(self.norm2(x))
        return x


# ── Architecture B: Hybrid Mamba + Attention ────────────────────────

class HybridParamEncoder(nn.Module):
    """Hybrid SSM+Attention: 2D CNN → [Mamba×5, Attn, Mamba, Attn] → DETR decoder → grouped heads."""

    def __init__(self, n_mels=128, d_model=256, d_state=64, nhead=8,
                 n_dec_layers=2, dropout=0.1):
        super().__init__()
        self.stem = CNN2DStem(d_model)
        self.pos_enc = LearnedPosEncoding2D(d_model, max_h=n_mels // 8, max_w=256 // 8)

        # 8 blocks: Mamba×5, Attention, Mamba, Attention
        self.blocks = nn.Sequential(
            MambaBlock(d_model, d_state, dropout=dropout),
            MambaBlock(d_model, d_state, dropout=dropout),
            MambaBlock(d_model, d_state, dropout=dropout),
            MambaBlock(d_model, d_state, dropout=dropout),
            MambaBlock(d_model, d_state, dropout=dropout),
            AttentionBlock(d_model, nhead, dropout),
            MambaBlock(d_model, d_state, dropout=dropout),
            AttentionBlock(d_model, nhead, dropout),
        )

        self.decoder = DETRDecoder(d_model, nhead, n_dec_layers)
        self.heads = GroupedParamHeads(d_model)
        self.d_model = d_model

    def forward(self, mel: torch.Tensor) -> torch.Tensor:
        x = self.stem(mel)
        seq_len = x.shape[1]
        h = 128 // 8
        w = seq_len // h
        pos = self.pos_enc(h, w).unsqueeze(0)
        if pos.shape[1] >= seq_len:
            x = x + pos[:, :seq_len]
        else:
            x = x + pos[:, :seq_len]

        memory = self.blocks(x)
        query_out = self.decoder(memory)
        return self.heads(query_out)

    def param_groups_for_optimizer(self, lr: float, lr_stem_factor: float = 0.3):
        return [
            {"params": self.stem.parameters(), "lr": lr * lr_stem_factor},
            {"params": list(self.blocks.parameters()) + list(self.pos_enc.parameters()), "lr": lr},
            {"params": list(self.decoder.parameters()) + list(self.heads.parameters()), "lr": lr},
        ]

from loom.training.dataset import N_CONTINUOUS, N_ROUTING, CATEGORICAL_KEYS

# Per-group loss weights (spec: osc=1, filter=2, env=3, fx=1)
_GROUP_WEIGHTS = None

def _get_group_weights(device):
    global _GROUP_WEIGHTS
    if _GROUP_WEIGHTS is not None and _GROUP_WEIGHTS.device == device:
        return _GROUP_WEIGHTS
    from loom.training.encoder_v2 import PARAM_GROUPS
    w = torch.ones(N_CONTINUOUS, device=device)
    for g in PARAM_GROUPS:
        for idx in g["cont"]:
            w[idx] = g["loss_weight"]
    _GROUP_WEIGHTS = w
    return w


def weighted_param_loss(pred: torch.Tensor, target: torch.Tensor) -> torch.Tensor:
    weights = _get_group_weights(pred.device)
    diff_sq = (pred[:, :N_CONTINUOUS] - target[:, :N_CONTINUOUS]).pow(2)
    continuous_mse = (diff_sq * weights.unsqueeze(0)).mean()

    cat_parts = []
    idx = N_CONTINUOUS
    for _, n in CATEGORICAL_KEYS:
        pred_logp = pred[:, idx:idx + n].clamp(1e-7, 1.0).log()
        target_p = target[:, idx:idx + n]
        cat_parts.append((-target_p * pred_logp).sum(dim=-1).mean())
        idx += n

    cat_ce = torch.stack(cat_parts).mean() if cat_parts else torch.tensor(0.0, device=pred.device)
    return continuous_mse + 0.5 * cat_ce
```

def train_epoch(model, train_mels, train_params, bs, optimizer, scaler, use_amp,
                synth, train_audio, args, ema_state):
    """Run one training epoch with gradient-accumulated spectral loss."""
    from loom.training.losses import weighted_param_loss
    model.train()
    n_train = len(train_mels)
    perm = torch.randperm(n_train, device=train_mels.device)
    loss_acc = 0.0
    n_batches = 0
    spec_sub_bs = args.spectral_batch_size

    for bi in range(0, n_train, bs):
        idx = perm[bi:bi + bs]
        mel = train_mels[idx]
        target = train_params[idx]

        with torch.amp.autocast("cuda", enabled=use_amp):
            pred = model(mel)

        pred = pred.float()
        loss_p = weighted_param_loss(pred, target)

        if synth is not None:
            # Spectral loss via gradient accumulation over sub-batches
            actual_bs = mel.shape[0]
            n_sub = max(1, (actual_bs + spec_sub_bs - 1) // spec_sub_bs)

            optimizer.zero_grad()
            loss_p.backward(retain_graph=True)
            gn_p = torch.stack([
                p.grad.norm() for p in model.parameters() if p.grad is not None
            ]).norm().item()
            saved_grads = [
                p.grad.detach().clone() for p in model.parameters()
                if p.grad is not None
            ]

            model.zero_grad()
            l_spectral_total = 0.0
            for si in range(0, actual_bs, spec_sub_bs):
                se = min(si + spec_sub_bs, actual_bs)
                pred_sub = pred[si:se]
                pred_p = vector_to_params(pred_sub)
                pred_p.pop("fx_routing", None)
                pred_audio, pred_inter = synth(pred_p, return_intermediates=True)

                with torch.no_grad():
                    target_sub = target[si:se]
                    target_p = vector_to_params(target_sub)
                    target_p.pop("fx_routing", None)
                    target_audio, target_inter = synth(target_p, return_intermediates=True)

                l_stft = multi_resolution_stft_loss(pred_audio, target_audio)
                l_chain = signal_chain_loss(pred_inter, target_inter)
                l_sub = (l_stft + l_chain) / n_sub
                l_sub.backward()
                l_spectral_total += l_sub.item()

            gn_s = torch.stack([
                p.grad.norm() for p in model.parameters() if p.grad is not None
            ]).norm().item()

            if ema_state["gn_param"] == 0.0:
                ema_state["gn_param"] = gn_p
                ema_state["gn_spectral"] = gn_s
            else:
                ema_state["gn_param"] = 0.99 * ema_state["gn_param"] + 0.01 * gn_p
                ema_state["gn_spectral"] = 0.99 * ema_state["gn_spectral"] + 0.01 * gn_s

            alpha = args.spectral_ratio * ema_state["gn_param"] / (ema_state["gn_spectral"] + 1e-8)
            gi = 0
            for p in model.parameters():
                if p.grad is not None:
                    p.grad.mul_(alpha).add_(saved_grads[gi])
                    gi += 1

            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            optimizer.step()
            loss_acc += loss_p.item() + alpha * l_spectral_total
        else:

            scaler.scale(loss_p).backward()
            scaler.unscale_(optimizer)
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            scaler.step(optimizer)
            scaler.update()
            loss_acc += loss_p.item()
        n_batches += 1

    return loss_acc / n_batches
